27// Define PRINTF_TIME and PRINTF_MICROSTEP, which are the printf
28// codes (like the d in %d to print an int) for time and microsteps.
29// To use these, specify the printf as follows:
30// printf("Time: " PRINTF_TIME "\n", time_value);
31// On most platforms, time is an signed 64-bit number (int64_t) and
32// the microstep is an unsigned 32-bit number (uint32_t).
33// Sadly, in C, there is no portable to print such numbers using
34// printf without getting a warning on some platforms.
35// On each platform, the code for printf if given by the macros
36// PRId64 and PRIu32 defined in inttypes.h. Hence, here, we import
37// inttypes.h, then define PRINTF_TIME and PRINTF_MICROSTEP.
38// If you are targeting a platform that uses some other type
39// for time and microsteps, you can simply define
40// PRINTF_TIME and PRINTF_MICROSTEP directly in the same file that
41// defines the types _instant_t, _interval_t, and _microstep_t.
42#include <inttypes.h>
43#define PRINTF_TIME "%" PRId64
44#define PRINTF_MICROSTEP "%" PRIu32
45
46// For convenience, the following string can be inserted in a printf
47// format for printing both time and microstep as follows:
48// printf("Tag is " PRINTF_TAG "\n", time_value, microstep);
49#define PRINTF_TAG "(" PRINTF_TIME ", " PRINTF_MICROSTEP ")"
