74static int lf_mutex_init(_lf_mutex_t* mutex) {
75 // Set up a recursive mutex
76 pthread_mutexattr_t attr;
77 pthread_mutexattr_init(&attr);
78 // Initialize the mutex to be recursive, meaning that it is OK
79 // for the same thread to lock and unlock the mutex even if it already holds
80 // the lock.
81 // FIXME: This is dangerous. The docs say this: "It is advised that an
82 // application should not use a PTHREAD_MUTEX_RECURSIVE mutex with
83 // condition variables because the implicit unlock performed for a
84 // pthread_cond_wait() or pthread_cond_timedwait() may not actually
85 // release the mutex (if it had been locked multiple times).
86 // If this happens, no other thread can satisfy the condition
87 // of the predicate.‚Äù This seems like a bug in the implementation of
88 // pthreads. Maybe it has been fixed?
89 pthread_mutexattr_settype(&attr, PTHREAD_MUTEX_RECURSIVE);
90 return pthread_mutex_init((pthread_mutex_t*)mutex, &attr);
91}

116static int lf_cond_init(_lf_cond_t* cond) {
117 pthread_condattr_t cond_attr;
118 pthread_condattr_init(&cond_attr);
119 // Limit the scope of the condition variable to this process (default)
120 pthread_condattr_setpshared(&cond_attr, PTHREAD_PROCESS_PRIVATE);
121 return pthread_cond_init(cond, &cond_attr);

160static int lf_cond_timedwait(_lf_cond_t* cond, _lf_mutex_t* mutex, int64_t absolute_time_ns) {
161 // Convert the absolute time to a timespec.
162 // timespec is seconds and nanoseconds.
163 struct timespec timespec_absolute_time
164 = {(time_t)absolute_time_ns / 1000000000LL, (long)absolute_time_ns % 1000000000LL};
165 int return_value = 0;
166 return_value = pthread_cond_timedwait(
167 (pthread_cond_t*)cond,
168 (pthread_mutex_t*)mutex,
169 &timespec_absolute_time
170 );
171 switch (return_value) {
172 case ETIMEDOUT:
173 return_value = _LF_TIMEOUT;
174 break;
175
176 default:
177 break;
178 }
179 return return_value;
180}
