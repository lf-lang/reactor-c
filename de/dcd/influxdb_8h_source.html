336{
337#define _APPEND(fmter...) \

339 if((written = snprintf(*buf + used, len - used, ##fmter)) < 0)\
340 goto FAIL;\

342 return -1;\
343 else {\
344 used += written;\
345 break;\
346 }\
347 }
348
349 size_t len = *_len;
350 int written = 0, type = 0, last_type = 0;
351 unsigned long long i = 0;
352 double d = 0.0;

356 used = 0;
357 }
358
359 type = va_arg(ap, int);
360 while(type != IF_TYPE_ARG_END) {
361 if(type >= IF_TYPE_TAG && type <= IF_TYPE_FIELD_BOOLEAN) {
362 if(last_type < IF_TYPE_MEAS || last_type > (type == IF_TYPE_TAG ? IF_TYPE_TAG : IF_TYPE_FIELD_BOOLEAN))
363 goto FAIL;
364 _APPEND("%c", (last_type <= IF_TYPE_TAG && type > IF_TYPE_TAG) ? ' ' : ',');
365 if(_escaped_append(buf, &len, &used, va_arg(ap, char*), ",= "))
366 return -2;
367 _APPEND("=");
368 }
369 switch(type) {
370 case IF_TYPE_MEAS:
371 if(last_type)
372 _APPEND("\n");
373 if(last_type && last_type <= IF_TYPE_TAG)
374 goto FAIL;
375 if(_escaped_append(buf, &len, &used, va_arg(ap, char*), ", "))
376 return -3;
377 break;
378 case IF_TYPE_TAG:
379 if(_escaped_append(buf, &len, &used, va_arg(ap, char*), ",= "))
380 return -4;
381 break;
382 case IF_TYPE_FIELD_STRING:
383 _APPEND("\"");
384 if(_escaped_append(buf, &len, &used, va_arg(ap, char*), "\""))
385 return -5;
386 _APPEND("\"");
387 break;
388 case IF_TYPE_FIELD_FLOAT:
389 d = va_arg(ap, double);
390 i = va_arg(ap, int);
391 _APPEND("%.*lf", (int)i, d);
392 break;
393 case IF_TYPE_FIELD_INTEGER:
394 i = va_arg(ap, long long);
395 _APPEND("%lldi", i);
396 break;
397 case IF_TYPE_FIELD_BOOLEAN:
398 i = va_arg(ap, int);
399 _APPEND("%c", i ? 't' : 'f');

401 case IF_TYPE_TIMESTAMP:

403 goto FAIL;
404 i = va_arg(ap, long long);
405 _APPEND(" %lld", i);
406 break;
407 default:
408 goto FAIL;
409 }

411 type = va_arg(ap, int);
412 }
413 _APPEND("\n");
414 if(last_type <= IF_TYPE_TAG)
415 goto FAIL;
416 *_len = len;
417 return used;
418FAIL:
