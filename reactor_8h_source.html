496struct trigger_t {
497 reaction_t** reactions; // Array of pointers to reactions sensitive to this trigger.
498 int number_of_reactions; // Number of reactions sensitive to this trigger.
499 bool is_timer; // True if this is a timer (a special kind of action), false otherwise.
500 interval_t offset; // Minimum delay of an action. For a timer, this is also the maximum delay.
501 interval_t period; // Minimum interarrival time of an action. For a timer, this is also the maximal interarrival time.
502 lf_token_t* token; // Pointer to a token wrapping the payload (or NULL if there is none).
503 bool is_physical; // Indicator that this denotes a physical action.
504 event_t* last; // Pointer to the last event that was scheduled for this action.
505 lf_spacing_policy_t policy; // Indicates which policy to use when an event is scheduled too early.
506 size_t element_size; // The size of the payload, if there is one, zero otherwise.
507 // If the payload is an array, then this is the size of an element of the array.
508 port_status_t status; // Determines the status of the port at the current logical time. Therefore, this
509 // value needs to be reset at the beginning of each logical time.
510 //
511 // This status is especially needed for the distributed execution because the receiver logic will need
512 // to know what it should do if it receives a message with 'intended tag = current tag' from another
513 // federate.
514 // - If status is 'unknown', it means that the federate has still no idea what the status of
515 // this port is and thus has refrained from executing any reaction that has that port as its input.
516 // This means that the receiver logic can directly inject the triggered reactions into the reaction
517 // queue at the current logical time.
518 // - If the status is absent, it means that the federate has assumed that the port is 'absent'
519 // for the current logical time. Therefore, receiving a message with 'intended tag = current tag'
520 // is an error that should be handled, for example, as a violation of the STP offset in the decentralized
521 // coordination.
522 // - Finally, if status is 'present', then this is an error since multiple
523 // downstream messages have been produced for the same port for the same logical time.
524#ifdef FEDERATED
525 tag_t last_known_status_tag; // Last known status of the port, either via a timed message, a port absent, or a
526 // TAG from the RTI.
527 bool is_a_control_reaction_waiting; // Indicates whether at least one control reaction is waiting for this trigger
528 // if it belongs to a network input port. Must be false by default.
529 tag_t intended_tag; // The amount of discrepency in logical time between the original intended
530 // trigger time of this trigger and the actual trigger time. This currently
531 // can only happen when logical connections are used using a decentralized coordination
532 // mechanism (@see https://github.com/icyphy/lingua-franca/wiki/Logical-Connections).
533 instant_t physical_time_of_arrival; // The physical time at which the message has been received on the network according to the local clock.
534 // Note: The physical_time_of_arrival is only passed down one level of the hierarchy. Default: NEVER.
535#endif
536};
